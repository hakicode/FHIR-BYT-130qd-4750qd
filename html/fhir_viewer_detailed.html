Thống Kê Số Giường

-- Chỉ số 1: Số BN nhập viện trong ngày
SoBNNhapVien =
CALCULATE(
    COUNTROWS(Encounter),
    FILTER(
        Encounter,
        Encounter[class_code] = "IMP"
        && INT(Encounter[period_start]) = INT(MAX(DimDate[Date]))
    )
)

-- Chỉ số 2: Số BN ra viện trong ngày
SoBNRaVien =
CALCULATE(
    COUNTROWS(Encounter),
    FILTER(
        Encounter,
        Encounter[class_code] = "IMP"
        && Encounter[status] = "finished"
        && NOT(ISBLANK(Encounter[period_end]))
        && INT(Encounter[period_end]) = INT(MAX(DimDate[Date]))
    )
)

-- Chỉ số 3: Số BN đang nằm viện (census)
SoBNDangNamVien =
CALCULATE(
    COUNTROWS(Encounter),
    FILTER(
        Encounter,
        Encounter[class_code] = "IMP"
        && Encounter[period_start] <= MAX(DimDate[Date])
        && (
            ISBLANK(Encounter[period_end])
            || Encounter[period_end] >= MAX(DimDate[Date])
        )
    )
)

-- Công suất sử dụng giường (kết hợp BC05)
CongSuatGiuong =
DIVIDE(
    [SoBNDangNamVien],
    SUM(Location[capacity]) * 1.0
)

-- Số hồ sơ ra viện theo ngày
SoHoSoRaVien =
CALCULATE(
    COUNTROWS(Encounter),
    Encounter[status] = "finished",
    NOT(ISBLANK(Encounter[period_end])),
    USERELATIONSHIP(DimDate[Date], Encounter[period_end_date])
)

-- Ngày ra viện (date only)
-- Calculated column trong Power Query / DAX:
period_end_date = DATE(
    YEAR(Encounter[period_end]), 
    MONTH(Encounter[period_end]), 
    DAY(Encounter[period_end])
)

-- Số ngày điều trị trung bình (BN ra viện trong ngày)
TBNgayDT_RaVien =
CALCULATE(
    AVERAGEX(
        Encounter,
        DATEDIFF(Encounter[period_start], Encounter[period_end], DAY)
    ),
    Encounter[status] = "finished",
    USERELATIONSHIP(DimDate[Date], Encounter[period_end_date])
)

-- Phân loại tình trạng ra viện
PhanLoaiRaVien =
SUMMARIZE(
    FILTER(Encounter, Encounter[status] = "finished"),
    Encounter[hospitalization_dischargeDisposition_code],
    Encounter[hospitalization_dischargeDisposition_display],
    "SoHS", COUNTROWS(Encounter)
)

-- Tỷ lệ tử vong (dischargeDisposition=exp)
TyLeTuVong =
DIVIDE(
    CALCULATE(COUNTROWS(Encounter), 
              Encounter[hospitalization_dischargeDisposition_code] = "exp"),
    COUNTROWS(Encounter)
)

-- Hồ sơ ngoại trú vs nội trú ra viện
HoSoTheoClass =
SUMMARIZE(
    FILTER(Encounter, Encounter[status] = "finished"),
    Encounter[class_code],
    "SoHS", COUNTROWS(Encounter),
    "TBNgay", AVERAGEX(Encounter, 
        DATEDIFF(Encounter[period_start], Encounter[period_end], DAY))
)
